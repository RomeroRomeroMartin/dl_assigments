def plot(history):
    accuracy = history.history['accuracy']
    val_accuracy = history.history['val_accuracy']
    loss = history.history['loss']
    val_loss = history.history['val_loss']

    epochs = range(1, len(accuracy) + 1)

    # Create subplots and unpack the axes array immediately
    fig, (ax_accuracy, ax_loss) = plt.subplots(1, 2, figsize=(20, 6))  # figsize=(width, height) in inches

    # Plot training and validation accuracy on the first axis
    ax_accuracy.plot(epochs, accuracy, 'b-o', label='Training accuracy')
    ax_accuracy.plot(epochs, val_accuracy, 'r-o', label='Validation accuracy')
    ax_accuracy.set_title('Training and Validation Accuracy')
    ax_accuracy.legend()
    ax_accuracy.set_xlabel('Epoch')
    ax_accuracy.set_ylabel('Accuracy')

    # Plot training and validation loss on the second axis
    ax_loss.plot(epochs, loss, 'b-o', label='Training loss')
    ax_loss.plot(epochs, val_loss, 'r-o', label='Validation loss')
    ax_loss.set_title('Training and Validation Loss')
    ax_loss.legend()
    ax_loss.set_xlabel('Epoch')
    ax_loss.set_ylabel('Loss')

    # Display the plot
    plt.show()

def save_model_results(file_path, model_name, accuracy):
    """
    This function saves the model's accuracy to a CSV file. It checks if the file exists
    to add a header if necessary and then opens the file in append mode to add
    the model name and accuracy.
    """
    # Check if the file exists to decide on adding the header
    header_needed = not os.path.isfile(file_path)
    
    # Open the file in append mode, which creates the file if it doesn't exist
    with open(file_path, "a") as file:
        # Add a header if the file was just created
        if header_needed:
            file.write("model_name,accuracy\n")
        # Append the model name and accuracy to the file
        file.write(f"{model_name},{accuracy:.3f}\n")

inputs = keras.Input(shape=img_tensor_shape)

x = layers.Rescaling(1./255)(inputs)

x = layers.Conv2D(filters=32, kernel_size=3, activation="relu")(x)
x = layers.MaxPooling2D(pool_size=2)(x)  # First MaxPooling layer

x = layers.Conv2D(filters=64, kernel_size=3, activation="relu")(x)
x = layers.MaxPooling2D(pool_size=2)(x)  # Second MaxPooling layer

x = layers.Conv2D(filters=128, kernel_size=3, activation="relu")(x)
x = layers.MaxPooling2D(pool_size=2)(x)  # Third MaxPooling layer

x = layers.Conv2D(filters=256, kernel_size=3, activation="relu")(x)

x = layers.Flatten()(x)  # Flatten the output to feed into a dense layer

# Adding some dense layers
x = layers.Dense(512, activation="relu")(x)
x = layers.Dropout(0.5)(x)  # Adding dropout for regularization
x = layers.Dense(256, activation="relu")(x)

outputs = layers.Dense(5, activation="softmax")(x)

model = keras.Model(inputs=inputs, outputs=outputs)

Discussion of Results¶

The results obtained from this model demonstrate a decent accuracy rate of 0.802 in validation. However, as evident from the plots, the model quickly begins to overfit the data. While the training loss shows improvement, the validation loss stagnate and gradually deteriorates. To address this, the next steps will involve developing more complex models but with stronger regularization techniques to prevent overfitting.

Some regularization - Dropout and batch-normalization¶

This model incorporates Dropout and Batch Normalization as key regularization techniques to enhance performance and prevent overfitting. Despite concerns that Dropout may interfere with Batch Normalization's effectiveness, their combined use has been shown to support better generalization in many successful models. Dropout reduces over-reliance on specific neurons by randomly disabling them during training, while Batch Normalization ensures stable learning by normalizing the inputs across mini-batches. Together, these strategies aim to improve the model's robustness and efficiency in learning from complex datasets.

def residual_block(x, filters, pooling=False):
    residual = x
    # First convolution without activation, followed by BatchNormalization and ReLU
    x = layers.Conv2D(filters, 3, padding="same")(x)
    x = layers.BatchNormalization()(x)
    x = layers.Activation("relu")(x)
    
    # Second convolution without activation, followed by BatchNormalization and ReLU
    x = layers.Conv2D(filters, 3, padding="same")(x)
    x = layers.BatchNormalization()(x)
    x = layers.Activation("relu")(x)
    
    if pooling:  # if pooling, we use strides in the residual connection
        x = layers.MaxPooling2D(2, padding="same")(x)
        residual = layers.Conv2D(filters, 1, strides=2)(residual)
    elif filters != residual.shape[-1]:
        # Adjusting the residual path's filter size if necessary
        residual = layers.Conv2D(filters, 1)(residual)
        residual = layers.BatchNormalization()(residual)  # Adding BatchNormalization here as well

    x = layers.add([x, residual])
    return x

def inception_module(x, filters_1x1, filters_3x3_reduce, filters_3x3, filters_5x5_reduce,
                           filters_5x5, filters_pool_proj, name=None):
    # 1x1 convolution followed by batch normalization
    conv_1x1 = layers.Conv2D(filters_1x1, (1, 1), padding='same', activation='relu')(x)
    conv_1x1 = layers.BatchNormalization()(conv_1x1)
    
    # 1x1 followed by 3x3 convolution, each followed by batch normalization
    pre_conv_3x3 = layers.Conv2D(filters_3x3_reduce, (1, 1), padding='same', activation='relu')(x)
    pre_conv_3x3 = layers.BatchNormalization()(pre_conv_3x3)
    conv_3x3 = layers.Conv2D(filters_3x3, (3, 3), padding='same', activation='relu')(pre_conv_3x3)
    conv_3x3 = layers.BatchNormalization()(conv_3x3)
    
    # 1x1 followed by 5x5 convolution, each followed by batch normalization
    pre_conv_5x5 = layers.Conv2D(filters_5x5_reduce, (1, 1), padding='same', activation='relu')(x)
    pre_conv_5x5 = layers.BatchNormalization()(pre_conv_5x5)
    conv_5x5 = layers.Conv2D(filters_5x5, (5, 5), padding='same', activation='relu')(pre_conv_5x5)
    conv_5x5 = layers.BatchNormalization()(conv_5x5)
    
    # Max pooling followed by 1x1 convolution, then batch normalization
    pool_proj = layers.MaxPooling2D((3, 3), strides=(1, 1), padding='same')(x)
    pool_proj = layers.Conv2D(filters_pool_proj, (1, 1), padding='same', activation='relu')(pool_proj)
    pool_proj = layers.BatchNormalization()(pool_proj)
    
    # Concatenate all the convolutions
    output = layers.concatenate([conv_1x1, conv_3x3, conv_5x5, pool_proj], axis=-1, name=name)
    
    return output
